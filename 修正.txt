原生微信小程序 · 本地存储 → 云端多用户协作

版本：v1.0 | 日期：2025年

本文档详细列出从单机本地存储版本升级为多用户实时协作版本的所有改动点，包含文件结构变更、接口规范、数据模型、权限设计、WebSocket协议及部署建议。

0. 改造总览

本次升级的核心目标是将完全离线的本地存储方案替换为基于云端 API 的多用户协作架构，同时保持小程序端的原生体验。主要涉及以下五个维度：

前端改造：utils 层网络请求封装、WebSocket 客户端、页面新增协作 UI

后端新建：四大服务（用户、日程、日历、协作）+ 实时推送服务

数据模型：原 localStorage 结构 → MySQL 关系型表设计

权限体系：基于角色的 RBAC（owner / editor / viewer）

实时同步：WebSocket 推送增量变更，解决多人并发编辑冲突

0.1 架构对比

4. WebSocket 实时协同协议

连接地址： wss://yourapi.com/ws?token=<JWT>，连接建立后服务端验证 token，失败则关闭连接（code 4001）。

4.1 客户端 → 服务端消息

type	payload 示例	说明
join_calendar	{ calendarId: "cal_001" }	订阅日历变更推送（切换页面时调用）
leave_calendar	{ calendarId: "cal_001" }	取消订阅（离开日历页面时调用）
editing_start	{ eventId: "evt_001" }	广播"我正在编辑"，让他人看到编辑状态
editing_stop	{ eventId: "evt_001" }	退出编辑（保存或取消后调用）
ping	{}	心跳检测，每25秒发一次
4.2 服务端 → 客户端推送

type	payload 结构 + 客户端处理方式
event_created	{ event: {...} } --- 直接插入本地列表，无需重新拉取
event_updated	{ eventId, changes: { title, startTime... }, updatedBy: { userId, nickname } } --- 差量更新对应条目
event_deleted	{ eventId, deletedBy } --- 从本地列表移除
member_joined	{ calendarId, user, permission } --- 刷新成员列表
member_removed	{ calendarId, userId } --- 若是自己被移除，退出该日历
someone_editing	{ eventId, user } --- 在日程卡片上显示"某人正在编辑"角标
someone_stopped	{ eventId } --- 移除编辑角标
notification	{ notif: {...} } --- 增加未读角标，必要时弹出 toast
pong	{} --- 重置心跳超时计时器
4.3 utils/ws.js 核心实现

// utils/ws.js
const app = getApp()

class WSClient {
constructor() {
this.socket = null
this.reconnectTimer = null
this.heartbeatTimer = null
this.listeners = {} // type -> [callback]
this.messageQueue = [] // 断线期间缓存的消息
}

connect() {
const token = wx.getStorageSync("token")
this.socket = wx.connectSocket({
url: wss://yourapi.com/ws?token=${token}
})
this.socket.onOpen(() => {
this._startHeartbeat()
this.messageQueue.forEach(msg => this.send(msg))
this.messageQueue = []
})
this.socket.onMessage(({ data }) => {
const msg = JSON.parse(data)
;(this.listeners[msg.type] || []).forEach(cb => cb(msg.payload))
})
this.socket.onClose(() => {
this._stopHeartbeat()
this.reconnectTimer = setTimeout(() => this.connect(), 3000)
})
}

send(type, payload = {}) {
const msg = JSON.stringify({ type, payload })
if (this.socket) this.socket.send({ data: msg })
else this.messageQueue.push({ type, payload })
}

on(type, cb) {
if (!this.listeners[type]) this.listeners[type] = []
this.listeners[type].push(cb)
}

_startHeartbeat() {
this.heartbeatTimer = setInterval(() => this.send("ping"), 25000)
}

_stopHeartbeat() { clearInterval(this.heartbeatTimer) }
}

export default new WSClient()

5.2 后端权限中间件实现

// middleware/permission.js
const { CalendarShare } = require("../models")

// 检查用户对某日历的最低权限
function requireCalendarPermission(minRole) {
return async (ctx, next) => {
const { calendarId } = ctx.params
const userId = ctx.state.userId // JWT 中间件注入

// owner 直接通过
const calendar = await Calendar.findByPk(calendarId)
if (calendar.ownerId === userId) return next()

// 查 share 表
const share = await CalendarShare.findOne({
where: { calendarId, userId, status: "accepted" }
})
if (!share) return ctx.throw(403, "无访问权限")
if (minRole === "edit" && share.permission !== "edit")
return ctx.throw(403, "需要编辑权限")

ctx.state.permission = share.permission
return next()
}
}

6. 并发冲突处理策略

6.1 乐观锁流程

编辑日程时采用乐观锁防止多人同时覆盖数据：

编辑前调用 POST /events/:id/lock，获取 lockToken（30秒有效）

服务端在 Redis 写入 lock:{eventId} = {lockToken, userId}，TTL 30s

PUT /events/:id 时携带 lockToken，服务端验证：

token 匹配 → 正常保存，version + 1，删除锁

token 不存在 → 锁已超时，返回 409 + 提示"锁已过期，请重新获取"

token 不匹配（他人加锁）→ 返回 409 + { lockedBy: { userId, nickname } }

前端30秒内无操作应自动续锁（重新调用 lock 接口），或释放锁

6.2 时间冲突检测 SQL

-- 检测 userId 在指定时间段内是否有其他日程冲突
SELECT e.event_id, e.title, e.start_time, e.end_time
FROM events e
INNER JOIN calendars c ON e.calendar_id = c.calendar_id
WHERE (c.owner_id = :userId
OR c.calendar_id IN (
SELECT calendar_id FROM calendar_shares
WHERE user_id = :userId AND status = "accepted"
))
AND e.event_id != :currentEventId
AND e.deleted_at IS NULL
AND e.start_time < :endTime
AND e.end_time > :startTime

7.1 utils/request.js（新建）

const BASE = "https://yourapi.com/api/v1"

export function request(method, path, data) {
return new Promise((resolve, reject) => {
const token = wx.getStorageSync("token")
wx.request({
url: BASE + path,
method,
data,
header: { "Authorization": Bearer ${token}, "Content-Type": "application/json" },
success: async (res) => {
if (res.statusCode === 200) return resolve(res.data)
if (res.statusCode === 401) {
// 自动续签
const newToken = await refreshToken()
if (newToken) return resolve(await request(method, path, data))
}
reject({ code: res.statusCode, ...res.data })
},
fail: reject
})
})
}

export const get = (path, params) => {
const qs = params ? "?" + new URLSearchParams(params).toString() : ""
return request("GET", path + qs)
}
export const post = (path, data) => request("POST", path, data)
export const put = (path, data) => request("PUT", path, data)
export const del = (path) => request("DELETE", path)

4.3 utils/ws.js 核心实现

// utils/ws.js
const app = getApp()

class WSClient {
constructor() {
this.socket = null
this.reconnectTimer = null
this.heartbeatTimer = null
this.listeners = {} // type -> [callback]
this.messageQueue = [] // 断线期间缓存的消息
}

connect() {
const token = wx.getStorageSync("token")
this.socket = wx.connectSocket({
url: wss://yourapi.com/ws?token=${token}
})
this.socket.onOpen(() => {
this._startHeartbeat()
this.messageQueue.forEach(msg => this.send(msg))
this.messageQueue = []
})
this.socket.onMessage(({ data }) => {
const msg = JSON.parse(data)
;(this.listeners[msg.type] || []).forEach(cb => cb(msg.payload))
})
this.socket.onClose(() => {
this._stopHeartbeat()
this.reconnectTimer = setTimeout(() => this.connect(), 3000)
})
}

send(type, payload = {}) {
const msg = JSON.stringify({ type, payload })
if (this.socket) this.socket.send({ data: msg })
else this.messageQueue.push({ type, payload })
}

on(type, cb) {
if (!this.listeners[type]) this.listeners[type] = []
this.listeners[type].push(cb)
}

_startHeartbeat() {
this.heartbeatTimer = setInterval(() => this.send("ping"), 25000)
}

_stopHeartbeat() { clearInterval(this.heartbeatTimer) }
}

export default new WSClient()

// 检查用户对某日历的最低权限
function requireCalendarPermission(minRole) {
return async (ctx, next) => {
const { calendarId } = ctx.params
const userId = ctx.state.userId // JWT 中间件注入

// owner 直接通过
const calendar = await Calendar.findByPk(calendarId)
if (calendar.ownerId === userId) return next()

// 查 share 表
const share = await CalendarShare.findOne({
where: { calendarId, userId, status: "accepted" }
})
if (!share) return ctx.throw(403, "无访问权限")
if (minRole === "edit" && share.permission !== "edit")
return ctx.throw(403, "需要编辑权限")

ctx.state.permission = share.permission
return next()
}
}

6. 并发冲突处理策略

6.1 乐观锁流程

编辑日程时采用乐观锁防止多人同时覆盖数据：

编辑前调用 POST /events/:id/lock，获取 lockToken（30秒有效）

服务端在 Redis 写入 lock:{eventId} = {lockToken, userId}，TTL 30s

PUT /events/:id 时携带 lockToken，服务端验证：

token 匹配 → 正常保存，version + 1，删除锁

token 不存在 → 锁已超时，返回 409 + 提示"锁已过期，请重新获取"

token 不匹配（他人加锁）→ 返回 409 + { lockedBy: { userId, nickname } }

前端30秒内无操作应自动续锁（重新调用 lock 接口），或释放锁

6.2 时间冲突检测 SQL

-- 检测 userId 在指定时间段内是否有其他日程冲突
SELECT e.event_id, e.title, e.start_time, e.end_time
FROM events e
INNER JOIN calendars c ON e.calendar_id = c.calendar_id
WHERE (c.owner_id = :userId
OR c.calendar_id IN (
SELECT calendar_id FROM calendar_shares
WHERE user_id = :userId AND status = "accepted"
))
AND e.event_id != :currentEventId
AND e.deleted_at IS NULL
AND e.start_time < :endTime
AND e.end_time > :startTime

const BASE = "https://yourapi.com/api/v1"

export function request(method, path, data) {
return new Promise((resolve, reject) => {
const token = wx.getStorageSync("token")
wx.request({
url: BASE + path,
method,
data,
header: { "Authorization": Bearer ${token}, "Content-Type": "application/json" },
success: async (res) => {
if (res.statusCode === 200) return resolve(res.data)
if (res.statusCode === 401) {
// 自动续签
const newToken = await refreshToken()
if (newToken) return resolve(await request(method, path, data))
}
reject({ code: res.statusCode, ...res.data })
},
fail: reject
})
})
}

export const get = (path, params) => {
const qs = params ? "?" + new URLSearchParams(params).toString() : ""
return request("GET", path + qs)
}
export const post = (path, data) => request("POST", path, data)
export const put = (path, data) => request("PUT", path, data)
export const del = (path) => request("DELETE", path)

8. 部署方案建议

8.1 方案一：微信云开发（推荐新手）

无需自建服务器，数据库、身份认证、存储全套托管，开发最快。

数据库：云数据库（MongoDB 风格）替代 MySQL，调整字段命名即可

身份认证：云函数内直接调用 getWXContext() 获取 openid，无需 JWT

WebSocket：微信云托管支持 WebSocket，或改用云函数 + 订阅消息替代实时推送

成本：基础版免费额度基本够用，月活万级以内几乎零成本

8.2 方案二：自建服务器（推荐有后端经验）

完全可控，本文档所有接口均基于此方案。

Runtime：Node.js 20 + Koa2 + Sequelize ORM

数据库：MySQL 8.0，Redis 7.0（Session + 乐观锁 + 缓存）

WebSocket：ws 库 + 自定义频道管理，或使用 Socket.IO

部署：Docker 容器化，Nginx 反代，腾讯云 / 阿里云轻量服务器 2C4G 足够

小程序合法域名：需在微信公众平台配置 API 域名和 WSS 域名白名单
